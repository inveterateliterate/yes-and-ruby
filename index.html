Improv Truth
Never deny the premise.

Block — flexible with args
def scene(offer, escalation)
  yield(offer, escalation)
end

offer = "I'm playing pickleball"
escalation = "heightening!"

scene(offer, escalation) do |line|
  puts line
end

# => "I'm playing pickleball"

Block — double time
def scene(offer, escalation)
  yield(offer)
  yield(nil, escalation)
end

offer = "I'm playing pickleball"
escalation = "and the ball is made of cheese!"

scene(offer, escalation) do |line, punchline|
  puts "#{line} #{punchline}"
end

# => "I'm playing pickleball"
# => "and the ball is made of cheese!"
